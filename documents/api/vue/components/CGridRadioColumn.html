# Vue Template Structure

<div class="demo-grid small">
  <c-grid
    :data="records"
    :frozen-col-count="1">
    <c-grid-radio-column
      field="c1"
      width="100"
    >
      Radio1
    </c-grid-radio-column>
    <c-grid-radio-column
      field="c2"
      width="100"
    >
      Radio2
    </c-grid-radio-column>
  </c-grid>
</div>
